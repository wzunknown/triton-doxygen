217  static void callbackBefore(triton::arch::Instruction* tritonInst, triton::uint8* addr, triton::uint32 size, CONTEXT* ctx, THREADID threadId) {
218  /* Some configurations must be applied before processing */
219  tracer::pintool::callbacks::preProcessing(tritonInst, threadId);
220 
221  if (!tracer::pintool::analysisTrigger.getState() || threadId != tracer::pintool::options::targetThreadId)
222  /* Analysis locked */
223  return;
224 
225  /* Mutex */
226  PIN_LockClient();
227 
228  /* Update CTX */
229  tracer::pintool::context::lastContext = ctx;
230 
231  /* Setup Triton information */
232  tritonInst->clear();
233  tritonInst->setOpcode(addr, size);
234  tritonInst->setAddress(reinterpret_cast<triton::__uint>(addr));
235  tritonInst->setThreadId(reinterpret_cast<triton::uint32>(threadId));
236 
237  /* Disassemble the instruction */
238  tracer::pintool::api.disassembly(*tritonInst);
239 
240  /* Execute the Python callback before the IR processing */
241  if (tracer::pintool::context::mustBeExecuted == false)
242  tracer::pintool::callbacks::beforeIRProc(tritonInst);
243  else
244  tracer::pintool::context::mustBeExecuted = false;
245 
246  /* Check if we must execute a new context */
247  if (tracer::pintool::context::mustBeExecuted == true) {
248  tritonInst->clear();
249  tracer::pintool::context::executeContext();
250  }
251 
252  /* Synchronize gliches between Pintool and libTriton */
253  tracer::pintool::context::synchronizeContext();
254 
255  /* Process the IR and spread taint only if one of both engines are enabled */
256  if (tracer::pintool::api.isTaintEngineEnabled() || tracer::pintool::api.isSymbolicEngineEnabled())
257  tracer::pintool::api.buildSemantics(*tritonInst);
258 
259  /* Execute the Python callback */
260  if (tracer::pintool::context::mustBeExecuted == false)
261  tracer::pintool::callbacks::before(tritonInst);
262 
263  /* Check if we must restore the snapshot */
264  if (tracer::pintool::snapshot.mustBeRestored() == true) {
265  tritonInst->clear();
266  tracer::pintool::snapshot.restoreSnapshot(ctx);
267  }
268 
269  /* Some configurations must be applied after processing */
270  tracer::pintool::callbacks::postProcessing(tritonInst, threadId);
271 
272  /* Mutex */

541  /* Check if the analysis must be unlocked */
542  static bool checkUnlockAnalysis(triton::__uint address) {
543  if (tracer::pintool::options::targetThreadId != -1)
544  return false;
545 
546  /* Unlock the analysis at the entry point from symbol */
547  if (tracer::pintool::options::startAnalysisFromSymbol != nullptr) {
548  if ((RTN_FindNameByAddress(address) == tracer::pintool::options::startAnalysisFromSymbol)) {
549  tracer::pintool::options::targetThreadId = PIN_ThreadId();
550  tracer::pintool::toggleWrapper(true);
551  return true;
552  }
553  }
554 
555  /* Unlock the analysis at the entry point from address */
556  else if (tracer::pintool::options::startAnalysisFromAddress.find(address) != tracer::pintool::options::startAnalysisFromAddress.end()) {
557  tracer::pintool::options::targetThreadId = PIN_ThreadId();
558  tracer::pintool::toggleWrapper(true);
559  return true;
560  }
561 
562  /* Unlock the analysis at the entry point from offset */
563  else if (tracer::pintool::options::startAnalysisFromOffset.find(tracer::pintool::getInsOffset(address)) != tracer::pintool::options::startAnalysisFromOffset.end()) {
564  tracer::pintool::options::targetThreadId = PIN_ThreadId();
565  tracer::pintool::toggleWrapper(true);
566  return true;
567  }
568  return false;

597  /* Trace instrumentation */
598  static void TRACE_Instrumentation(TRACE trace, VOID *v) {
599 
600  for (BBL bbl = TRACE_BblHead(trace); BBL_Valid(bbl); bbl = BBL_Next(bbl)) {
601  for (INS ins = BBL_InsHead(bbl); INS_Valid(ins); ins = INS_Next(ins)) {
602 
603  /* Check if the analysis me be unlocked */
604  tracer::pintool::checkUnlockAnalysis(INS_Address(ins));
605 
606  if (!tracer::pintool::analysisTrigger.getState())
607  /* Analysis locked */
608  continue;
609 
610  if (tracer::pintool::instructionBlacklisted(INS_Address(ins)) == true || tracer::pintool::instructionWhitelisted(INS_Address(ins)) == false)
611  /* Insruction blacklisted */
612  continue;
613 
614  /* Prepare the Triton's instruction */
615  triton::arch::Instruction* tritonInst = new triton::arch::Instruction();
616 
617  /* Save memory read1 informations */
618  if (INS_IsMemoryRead(ins)) {
619  INS_InsertCall(ins, IPOINT_BEFORE, (AFUNPTR)saveMemoryAccess,
620  IARG_PTR, tritonInst,
621  IARG_MEMORYREAD_EA,
622  IARG_MEMORYREAD_SIZE,
623  IARG_END);
624  }
625 
626  /* Save memory read2 informations */
627  if (INS_HasMemoryRead2(ins)) {
628  INS_InsertCall(ins, IPOINT_BEFORE, (AFUNPTR)saveMemoryAccess,
629  IARG_PTR, tritonInst,
630  IARG_MEMORYREAD2_EA,
631  IARG_MEMORYREAD_SIZE,
632  IARG_END);
633  }
634 
635  /* Callback before */
636  INS_InsertCall(ins, IPOINT_BEFORE, (AFUNPTR)callbackBefore,
637  IARG_PTR, tritonInst,
638  IARG_INST_PTR,
639  IARG_UINT32, INS_Size(ins),
640  IARG_CONTEXT,
641  IARG_THREAD_ID,
642  IARG_END);
643 
644  /* Callback after */
645  /* Syscall after context must be catcher with INSERT_POINT.SYSCALL_EXIT */
646  if (INS_IsSyscall(ins) == false) {
647  IPOINT where = IPOINT_AFTER;
648  if (INS_HasFallThrough(ins) == false)
649  where = IPOINT_TAKEN_BRANCH;
650  INS_InsertCall(ins, where, (AFUNPTR)callbackAfter, IARG_PTR, tritonInst, IARG_CONTEXT, IARG_THREAD_ID, IARG_END);
651  }
652 
653  /* I/O memory monitoring for snapshot */
654  if (INS_OperandCount(ins) > 1 && INS_MemoryOperandIsWritten(ins, 0)) {
655  INS_InsertCall(
656  ins, IPOINT_BEFORE, (AFUNPTR)callbackSnapshot,
657  IARG_MEMORYOP_EA, 0,
658  IARG_UINT32, INS_MemoryWriteSize(ins),
659  IARG_END);
660  }
661 
662  }
663  }
