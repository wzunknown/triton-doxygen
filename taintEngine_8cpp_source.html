76  /* Returns true of false if the memory address is currently tainted */
77  bool TaintEngine::isMemoryTainted(const triton::arch::MemoryAccess& mem, bool mode) const {
78  triton::uint64 addr = mem.getAddress();
79  triton::uint32 size = mem.getSize();
80 
81  for (triton::uint32 index = 0; index < size; index++) {
82  if (this->taintedMemory.find(addr+index) != this->taintedMemory.end())
83  return TAINTED;
84  }
85 
86  /* Spread the taint through pointers if the mode is enabled */
87  if (mode && this->modes->isModeEnabled(triton::modes::TAINT_THROUGH_POINTERS)) {
88  if (this->isRegisterTainted(mem.getConstBaseRegister()))
89  return TAINTED;
90  if (this->isRegisterTainted(mem.getConstIndexRegister()))
91  return TAINTED;
92  if (this->isRegisterTainted(mem.getConstSegmentRegister()))
93  return TAINTED;
94  }
95 
96  return !TAINTED;
97  }
