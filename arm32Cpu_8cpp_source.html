239  void Arm32Cpu::disassembly(triton::arch::Instruction& inst) const {
240  triton::extlibs::capstone::csh handle;
241  triton::extlibs::capstone::cs_insn* insn;
242  triton::usize count = 0;
243 
244  /* Check if the opcode and opcode' size are defined */
245  if (inst.getOpcode() == nullptr || inst.getSize() == 0)
246  throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Opcode and opcodeSize must be definied.");
247 
248  /* Select capstone handler (based on execution mode) */
249  handle = (this->thumb ? this->handle_thumb : this->handle_arm);
250 
251  /* Clear instructicon's operands if alredy defined */
252  inst.operands.clear();
253 
254  /* Update instruction address if undefined */
255  if (!inst.getAddress()) {
256  inst.setAddress(this->getConcreteRegisterValue(this->getProgramCounter()).convert_to<triton::uint64>());
257  }
258 
259  /* Let's disass and build our operands */
260  count = triton::extlibs::capstone::cs_disasm(handle, inst.getOpcode(), inst.getSize(), inst.getAddress(), 0, &insn);
261  if (count > 0) {
262  triton::extlibs::capstone::cs_detail* detail = insn->detail;
263  for (triton::uint32 j = 0; j < 1; j++) {
264 
265  /* Init the disassembly */
266  std::stringstream str;
267 
268  /* Add mnemonic */
269  str << insn[j].mnemonic;
270 
271  /* Add operands */
272  if (detail->arm.op_count)
273  str << " " << insn[j].op_str;
274 
275  inst.setDisassembly(str.str());
276 
277  /* Refine the opcode */
278  inst.setOpcode(insn[j].bytes, insn[j].size);
279 
280  /* Refine the size */
281  inst.setSize(insn[j].size);
282 
283  /* Init the instruction's type */
284  inst.setType(this->capstoneInstructionToTritonInstruction(insn[j].id));
285 
286  /* Init the instruction's code codition */
287  inst.setCodeCondition(this->capstoneConditionToTritonCondition(detail->arm.cc));
288 
289  /* Init the instruction's write back flag */
290  inst.setWriteBack(detail->arm.writeback);
291 
292  /* Set True if the instruction udpate flags */
293  inst.setUpdateFlag(detail->arm.update_flags);
294 
295  /* Set thumb mode */
296  inst.setThumb(thumb);
297 
298  /* Init operands */
299  for (triton::uint32 n = 0; n < detail->arm.op_count; n++) {
300  triton::extlibs::capstone::cs_arm_op* op = &(detail->arm.operands[n]);
301  switch(op->type) {
302 
303  case triton::extlibs::capstone::ARM_OP_IMM: {
304  triton::arch::Immediate imm(op->imm, triton::size::dword);
305 
306  if (op->subtracted)
307  imm.setSubtracted(true);
308 
309  inst.operands.push_back(triton::arch::OperandWrapper(imm));
310  break;
311  }
312 
313  case triton::extlibs::capstone::ARM_OP_MEM: {
314  triton::arch::MemoryAccess mem;
315 
316  /* Set the size of the memory access */
317  mem.setBits(triton::bitsize::dword-1, 0);
318 
319  /* LEA if exists */
320  const triton::arch::Register base(*this, this->capstoneRegisterToTritonRegister(op->mem.base));
321  triton::arch::Register index(*this, this->capstoneRegisterToTritonRegister(op->mem.index));
322 
323  /* Set Shift type and value */
324  triton::arch::arm::shift_e shiftType = this->capstoneShiftToTritonShift(op->shift.type);
325 
326  index.setShiftType(shiftType);
327 
328  switch(shiftType) {
329  case triton::arch::arm::ID_SHIFT_INVALID:
330  break;
331  case triton::arch::arm::ID_SHIFT_ASR:
332  case triton::arch::arm::ID_SHIFT_LSL:
333  case triton::arch::arm::ID_SHIFT_LSR:
334  case triton::arch::arm::ID_SHIFT_ROR:
335  index.setShiftValue(op->shift.value);
336  break;
337  case triton::arch::arm::ID_SHIFT_RRX:
338  /* NOTE: According to the manual RRX there is no
339  * immediate associated with this shift type. However,
340  * from the description of the instruction it can be
341  * deduced that a value of one is used.
342  */
343  index.setShiftValue(1);
344  break;
345  case triton::arch::arm::ID_SHIFT_ASR_REG:
346  case triton::arch::arm::ID_SHIFT_LSL_REG:
347  case triton::arch::arm::ID_SHIFT_LSR_REG:
348  case triton::arch::arm::ID_SHIFT_ROR_REG:
349  index.setShiftValue(this->capstoneRegisterToTritonRegister(op->shift.value));
350  break;
351  case triton::arch::arm::ID_SHIFT_RRX_REG:
352  /* NOTE: Capstone considers this as a viable shift operand
353  * but according to the ARM manual this is not possible.
354  */
355  throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Invalid shift type.");
356  break;
357  default:
358  throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Invalid shift type.");
359  }
360 
361  if (op->subtracted) {
362  index.setSubtracted(true);
363  }
364 
365  triton::uint32 immsize = (
366  this->isRegisterValid(base.getId()) ? base.getSize() :
367  this->isRegisterValid(index.getId()) ? index.getSize() :
368  this->gprSize()
369  );
370 
371  triton::arch::Immediate disp(op->mem.disp, immsize);
372 
373  /* Specify that LEA contains a PC relative */
374  if (base.getId() == this->pcId) {
375  /* NOTE: PC always points to the address to the current
376  * instruction plus: a) 8 in case of ARM mode, or b) 4 in
377  * case of Thumb. It is also aligned to 4 bytes. For more
378  * information, refer to section "Use of labels in UAL
379  * instruction syntax" of the reference manual.
380  */
381  auto offset = this->thumb ? 4 : 8;
382  auto address = (inst.getAddress() + offset) & 0xfffffffc;
383  mem.setPcRelative(address);
384  }
385 
386  /* Note that in ARM32 there is no segment register and scale value */
387  mem.setBaseRegister(base);
388  mem.setIndexRegister(index);
389  mem.setDisplacement(disp);
390 
391  /* If there is an index register available, set scale to 1 to perform this following computation (base) + (index * scale) */
392  if (this->isRegisterValid(index.getId()))
393  mem.setScale(triton::arch::Immediate(1, index.getSize()));
394 
395  inst.operands.push_back(triton::arch::OperandWrapper(mem));
396  break;
397  }
398 
399  case triton::extlibs::capstone::ARM_OP_REG: {
400  triton::arch::Register reg(*this, this->capstoneRegisterToTritonRegister(op->reg));
401 
402  /* Set Shift type and value */
403  triton::arch::arm::shift_e shiftType = this->capstoneShiftToTritonShift(op->shift.type);
404 
405  reg.setShiftType(shiftType);
406 
407  switch(shiftType) {
408  case triton::arch::arm::ID_SHIFT_INVALID:
409  break;
410  case triton::arch::arm::ID_SHIFT_ASR:
411  case triton::arch::arm::ID_SHIFT_LSL:
412  case triton::arch::arm::ID_SHIFT_LSR:
413  case triton::arch::arm::ID_SHIFT_ROR:
414  reg.setShiftValue(op->shift.value);
415  break;
416  case triton::arch::arm::ID_SHIFT_RRX:
417  /* NOTE: According to the manual RRX there is no
418  * immediate associated with this shift type. However,
419  * from the description of the instruction it can be
420  * deduced that a value of one is used.
421  */
422  reg.setShiftValue(1);
423  break;
424  case triton::arch::arm::ID_SHIFT_ASR_REG:
425  case triton::arch::arm::ID_SHIFT_LSL_REG:
426  case triton::arch::arm::ID_SHIFT_LSR_REG:
427  case triton::arch::arm::ID_SHIFT_ROR_REG:
428  reg.setShiftValue(this->capstoneRegisterToTritonRegister(op->shift.value));
429  break;
430  case triton::arch::arm::ID_SHIFT_RRX_REG:
431  /* NOTE: Capstone considers this as a viable shift operand
432  * but according to the ARM manual this is not possible.
433  */
434  throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Invalid shift type.");
435  break;
436  default:
437  throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Invalid shift type.");
438  }
439 
440  if (op->subtracted)
441  reg.setSubtracted(true);
442 
443  inst.operands.push_back(triton::arch::OperandWrapper(reg));
444  break;
445  }
446 
447  default:
448  /* NOTE: FP, CIMM, and missing one are not supported yet. */
449  throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Invalid operand.");
450  } // switch
451  } // for operand
452  } // for instruction
453 
454  /* Set branch */
455  if (detail->groups_count > 0) {
456  for (triton::uint32 n = 0; n < detail->groups_count; n++) {
457  if (detail->groups[n] == triton::extlibs::capstone::ARM_GRP_JUMP) {
458  inst.setBranch(true);
459  inst.setControlFlow(true);
460  }
461  }
462  }
463 
464  /* Post process instruction */
465  this->postDisassembly(inst);
466 
467  /* Free capstone stuffs */
468  triton::extlibs::capstone::cs_free(insn, count);
469  }
470  else
471  throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Failed to disassemble the given code.");
472 
473  return;
474  }

477  void Arm32Cpu::postDisassembly(triton::arch::Instruction& inst) const {
478  /* Fix update flag */
479  /* NOTE: Quick (and super ugly) hack. Capstone is reporting
480  * update_flags equals true for ADC, RSC and SBC instruction when
481  * it shouldn't (it should only report true when the S suffix is
482  * present). This will be removed once Capstone is fixed. For more
483  * information, see: https://github.com/aquynh/capstone/issues/1568.
484  */
485  switch (inst.getType()) {
486  case ID_INS_ADC:
487  case ID_INS_RSC:
488  case ID_INS_SBC:
489  if (inst.getDisassembly().at(3) != 's')
490  inst.setUpdateFlag(false);
491  break;
492  }
493 
494  /* Make implicit destination operand explicit */
495  /* NOTE: For some instructions the destination operand is
496  * optional (in which case the first source operand is used as
497  * destination). Capstone returns always all three operands for
498  * ARM instruction (i.e. make the destination operand explicit).
499  * However, it does not do the same for Thumb instructions. Here
500  * we make the destination operand explicit (in order to simplify
501  * the implementation of the semantics).
502  */
503  if (inst.isThumb() && inst.operands.size() == 2) {
504  triton::arch::OperandWrapper op(inst.operands[0]);
505 
506  switch (inst.getType()) {
507  case ID_INS_ADC:
508  case ID_INS_ADD:
509  case ID_INS_AND:
510  case ID_INS_ASR:
511  case ID_INS_BIC:
512  case ID_INS_EOR:
513  case ID_INS_LSL:
514  case ID_INS_LSR:
515  case ID_INS_ORR:
516  case ID_INS_ROR:
517  case ID_INS_SBC:
518  case ID_INS_SUB:
519  inst.operands.insert(inst.operands.begin(), op);
520  break;
521  }
522  }
523 
524  /* NOTE: If the instruction is POP and contains a PC register,
525  * we have to define the instruction as modifiying the control
526  * flow. See #945.
527  */
528  if (inst.getType() == ID_INS_POP) {
529  /* FIXME: Maybe the loop is useless if PC is always the last operand? */
530  for (auto& op : inst.operands) {
531  if (op.getType() == triton::arch::OP_REG && op.getConstRegister().getId() == this->pcId) {
532  inst.setControlFlow(true);
533  break;
534  }
535  }
536  }
537  }

589  switch (reg.getId()) {

591  case triton::arch::ID_REG_ARM32_R1: return (*((triton::uint32*)(this->r1)));
592  case triton::arch::ID_REG_ARM32_R2: return (*((triton::uint32*)(this->r2)));

626  void Arm32Cpu::setConcreteMemoryValue(const triton::arch::MemoryAccess& mem, const triton::uint512& value) {
627  triton::uint64 addr = mem.getAddress();
628  triton::uint32 size = mem.getSize();
629  triton::uint512 cv = value;
630 
631  if (cv > mem.getMaxValue())
632  throw triton::exceptions::Register("Arm32Cpu::setConcreteMemoryValue(): You cannot set this concrete value (too big) to this memory access.");
633 
634  if (size == 0 || size > triton::size::dqqword)
635  throw triton::exceptions::Cpu("Arm32Cpu::setConcreteMemoryValue(): Invalid size memory.");
636 
637  if (this->callbacks)
638  this->callbacks->processCallbacks(triton::callbacks::SET_CONCRETE_MEMORY_VALUE, mem, value);
639 
640  for (triton::uint32 i = 0; i < size; i++) {
641  this->memory[addr+i] = (cv & 0xff).convert_to<triton::uint8>();
642  cv >>= 8;
643  }

663  void Arm32Cpu::setConcreteRegisterValue(const triton::arch::Register& reg, const triton::uint512& value) {
664  if (value > reg.getMaxValue())
665  throw triton::exceptions::Register("Arm32Cpu::setConcreteRegisterValue(): You cannot set this concrete value (too big) to this register.");
666 
667  if (this->callbacks)
668  this->callbacks->processCallbacks(triton::callbacks::SET_CONCRETE_REGISTER_VALUE, reg, value);
669 
670  switch (reg.getId()) {
671  case triton::arch::ID_REG_ARM32_R0: (*((triton::uint32*)(this->r0))) = value.convert_to<triton::uint32>(); break;
672  case triton::arch::ID_REG_ARM32_R1: (*((triton::uint32*)(this->r1))) = value.convert_to<triton::uint32>(); break;
673  case triton::arch::ID_REG_ARM32_R2: (*((triton::uint32*)(this->r2))) = value.convert_to<triton::uint32>(); break;
674  case triton::arch::ID_REG_ARM32_R3: (*((triton::uint32*)(this->r3))) = value.convert_to<triton::uint32>(); break;
675  case triton::arch::ID_REG_ARM32_R4: (*((triton::uint32*)(this->r4))) = value.convert_to<triton::uint32>(); break;
676  case triton::arch::ID_REG_ARM32_R5: (*((triton::uint32*)(this->r5))) = value.convert_to<triton::uint32>(); break;
677  case triton::arch::ID_REG_ARM32_R6: (*((triton::uint32*)(this->r6))) = value.convert_to<triton::uint32>(); break;
678  case triton::arch::ID_REG_ARM32_R7: (*((triton::uint32*)(this->r7))) = value.convert_to<triton::uint32>(); break;
679  case triton::arch::ID_REG_ARM32_R8: (*((triton::uint32*)(this->r8))) = value.convert_to<triton::uint32>(); break;
680  case triton::arch::ID_REG_ARM32_R9: (*((triton::uint32*)(this->r9))) = value.convert_to<triton::uint32>(); break;
681  case triton::arch::ID_REG_ARM32_R10: (*((triton::uint32*)(this->r10))) = value.convert_to<triton::uint32>(); break;
682  case triton::arch::ID_REG_ARM32_R11: (*((triton::uint32*)(this->r11))) = value.convert_to<triton::uint32>(); break;
683  case triton::arch::ID_REG_ARM32_R12: (*((triton::uint32*)(this->r12))) = value.convert_to<triton::uint32>(); break;
684  case triton::arch::ID_REG_ARM32_SP: (*((triton::uint32*)(this->sp))) = value.convert_to<triton::uint32>(); break;
685  case triton::arch::ID_REG_ARM32_R14: (*((triton::uint32*)(this->r14))) = value.convert_to<triton::uint32>(); break;
686  case triton::arch::ID_REG_ARM32_PC: {
687  /* NOTE: Once in Thumb mode only switch to ARM through a Branch
688  * and Exchange instruction. The reason for this is that after
689  * switching to Thumb the ISB (instruction set selection bit) is
690  * cleared. Therefore, if we allow to switch back to ARM through
691  * these mechanism we would have a problem processing Thumb
692  * instructions.
693  */
694  auto pc = value.convert_to<triton::uint32>();
695  if (this->isThumb() == false && (pc & 0x1) == 0x1) {
696  this->setThumb(true);
697  }
698  (*((triton::uint32*)(this->pc))) = pc & ~0x1;
699  break;
700  }
701  case triton::arch::ID_REG_ARM32_APSR: (*((triton::uint32*)(this->apsr))) = value.convert_to<triton::uint32>(); break;
702  case triton::arch::ID_REG_ARM32_N: {
703  triton::uint32 b = (*((triton::uint32*)(this->apsr)));
704  (*((triton::uint32*)(this->apsr))) = !value.is_zero() ? b | (1 << 31) : b & ~(1 << 31);
705  break;
706  }
707  case triton::arch::ID_REG_ARM32_Z: {
708  triton::uint32 b = (*((triton::uint32*)(this->apsr)));
709  (*((triton::uint32*)(this->apsr))) = !value.is_zero() ? b | (1 << 30) : b & ~(1 << 30);
710  break;
711  }
712  case triton::arch::ID_REG_ARM32_C: {
713  triton::uint32 b = (*((triton::uint32*)(this->apsr)));
714  (*((triton::uint32*)(this->apsr))) = !value.is_zero() ? b | (1 << 29) : b & ~(1 << 29);
715  break;
716  }
717  case triton::arch::ID_REG_ARM32_V: {
718  triton::uint32 b = (*((triton::uint32*)(this->apsr)));
719  (*((triton::uint32*)(this->apsr))) = !value.is_zero() ? b | (1 << 28) : b & ~(1 << 28);
720  break;
721  }
722  default:
723  throw triton::exceptions::Cpu("Arm32Cpu:setConcreteRegisterValue(): Invalid register.");
724  }
725  }
