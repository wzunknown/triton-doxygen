65  /* Restore the snapshot. */
66  void Snapshot::restoreSnapshot(CONTEXT *ctx) {
67 
68  if (this->mustBeRestore == false)
69  return;
70 
71  /* 1 - Restore all memory modification. */
72  for(auto i = this->memory.begin(); i != this->memory.end(); ++i){
73  *(reinterpret_cast<char*>(i->first)) = i->second;
74  }
75  this->memory.clear();
76 
77  /* 2 - Restore current symbolic engine state */
78  *tracer::pintool::api.getSymbolicEngine() = *this->snapshotSymEngine;
79 
80  /* 3 - Restore current taint engine state */
81  *tracer::pintool::api.getTaintEngine() = *this->snapshotTaintEngine;
82 
83  /* 4 - Restore AST Context */
84  *tracer::pintool::api.getAstContext() = *this->astCtx;
85 
86  /* 5 - Restore the Triton CPU state */
87  #if defined(__x86_64__) || defined(_M_X64)
88  *dynamic_cast<triton::arch::x86::x8664Cpu*>(tracer::pintool::api.getCpuInstance()) = *this->cpu;
89  #endif
90  #if defined(__i386) || defined(_M_IX86)
91  *dynamic_cast<triton::arch::x86::x86Cpu*>(tracer::pintool::api.getCpuInstance()) = *this->cpu;
92  #endif
93 
94  /* 6 - Restore Pin registers context */
95  PIN_SaveContext(&this->pinCtx, ctx);
96 
97  this->mustBeRestore = false;
98  PIN_UnlockClient();
