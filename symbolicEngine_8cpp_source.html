175  /* Removes an aligned memory */
176  void SymbolicEngine::removeAlignedMemory(triton::uint64 address, triton::uint32 size) {
177  /*
178  * Avoid accessing the alignedMemoryReference array when empty. This usually happens when
179  * you initialize the symbolic engine and concretize whole sections of an executable using
180  * setConcreteMemoryValue. No symbolic memory has been created yet but this function will
181  * still try to rougly erase (size * 7) elements.
182  */
183  if (this->alignedMemoryReference.empty())
184  return;
185 
186  /* Remove overloaded positive ranges */
187  for (triton::uint32 index = 0; index < size; index++) {
188  this->alignedMemoryReference.erase(std::make_pair(address+index, triton::size::byte));
189  this->alignedMemoryReference.erase(std::make_pair(address+index, triton::size::word));
190  this->alignedMemoryReference.erase(std::make_pair(address+index, triton::size::dword));
191  this->alignedMemoryReference.erase(std::make_pair(address+index, triton::size::qword));
192  this->alignedMemoryReference.erase(std::make_pair(address+index, triton::size::dqword));
193  this->alignedMemoryReference.erase(std::make_pair(address+index, triton::size::qqword));
194  this->alignedMemoryReference.erase(std::make_pair(address+index, triton::size::dqqword));
195  }
196 
197  /* Remove overloaded negative ranges */
198  for (triton::uint32 index = 1; index < triton::size::dqqword; index++) {
199  if (index < triton::size::word) this->alignedMemoryReference.erase(std::make_pair(address-index, triton::size::word));
200  if (index < triton::size::dword) this->alignedMemoryReference.erase(std::make_pair(address-index, triton::size::dword));
201  if (index < triton::size::qword) this->alignedMemoryReference.erase(std::make_pair(address-index, triton::size::qword));
202  if (index < triton::size::dqword) this->alignedMemoryReference.erase(std::make_pair(address-index, triton::size::dqword));
203  if (index < triton::size::qqword) this->alignedMemoryReference.erase(std::make_pair(address-index, triton::size::qqword));
204  if (index < triton::size::dqqword) this->alignedMemoryReference.erase(std::make_pair(address-index, triton::size::dqqword));
205  }

480  std::ostream& SymbolicEngine::printSlicedExpressions(std::ostream& stream, const SharedSymbolicExpression& expr, bool assert_) {

483  std::vector<triton::usize> symExprs;
484  for (const auto& se : ssa) {
485  symExprs.push_back(se.first);

490  for (const auto& n : ast::search(expr->getAst(), ast::VARIABLE_NODE)) {

492  symVars[var->getId()] = var;

496  for (const auto& var : symVars) {
497  auto n = this->astCtxt->declare(this->astCtxt->variable(var.second));
498  this->astCtxt->print(stream, n.get());

503  std::sort(symExprs.begin(), symExprs.end());
504  if (assert_) {
505  symExprs.pop_back();
506  }
507 
508  /* Print symbolic expressions */
509  for (const auto& id : symExprs) {
510  stream << ssa[id]->getFormattedExpression() << std::endl;
511  }
512 
513  if (assert_) {
514  /* Print conjuncts in separate asserts */
515  std::vector<ast::SharedAbstractNode> exprs;
516  std::vector<ast::SharedAbstractNode> wl{expr->getAst()};
517 
518  while (!wl.empty()) {
519  auto n = wl.back();
520  wl.pop_back();
521 
522  if (n->getType() != ast::LAND_NODE) {
523  exprs.push_back(n);
524  continue;
525  }
526 
527  for (const auto& child : n->getChildren()) {
528  wl.push_back(child);
529  }
530  }
531 
532  for (auto it = exprs.crbegin(); it != exprs.crend(); ++it) {
533  this->astCtxt->print(stream, this->astCtxt->assert_(*it).get());
534  stream << std::endl;
535  }
536 
537  if (this->astCtxt->getRepresentationMode() == ast::representations::SMT_REPRESENTATION) {
538  stream << "(check-sat)" << std::endl;
539  stream << "(get-model)" << std::endl;
540  }
541  }

619  const SharedSymbolicVariable& symVar = this->newSymbolicVariable(MEMORY_VARIABLE, memAddr, symVarSize * bitsize::byte, symVarAlias);
620 
621  /* Create the AST node */
622  const triton::ast::SharedAbstractNode& symVarNode = this->astCtxt->variable(symVar);
623 
624  /* Setup the concrete value to the symbolic variable */
625  this->setConcreteVariableValue(symVar, cv);
626 
627  /* Record the aligned symbolic variable for a symbolic optimization */
628  if (this->modes->isModeEnabled(triton::modes::ALIGNED_MEMORY)) {
629  const SharedSymbolicExpression& se = this->newSymbolicExpression(symVarNode, MEMORY_EXPRESSION, "aligned Byte reference");
630  se->setOriginMemory(mem);
631  this->addAlignedMemory(memAddr, symVarSize, se);
632  }
633 
634  /* Split expression in bytes */
635  for (triton::sint32 index = symVarSize-1; index >= 0; index--) {
636  triton::uint32 high = ((bitsize::byte * (index + 1)) - 1);
637  triton::uint32 low = ((bitsize::byte * (index + 1)) - bitsize::byte);
638 
639  /* Isolate the good part of the symbolic variable */
640  const triton::ast::SharedAbstractNode& tmp = this->astCtxt->extract(high, low, symVarNode);
641 
642  /* Create a new symbolic expression containing the symbolic variable */
643  const SharedSymbolicExpression& se = this->newSymbolicExpression(tmp, MEMORY_EXPRESSION, "Byte reference");
644  se->setOriginMemory(triton::arch::MemoryAccess(memAddr+index, triton::size::byte));
645 
646  /* Assign the symbolic expression to the memory cell */
647  this->addMemoryReference(memAddr+index, se);
648  }
649 
650  return symVar;
651  }

871  /* Returns the AST corresponding to the memory */
872  triton::ast::SharedAbstractNode SymbolicEngine::getMemoryAst(const triton::arch::MemoryAccess& mem) {
873  std::vector<triton::ast::SharedAbstractNode> opVec;
874 
875  triton::ast::SharedAbstractNode tmp = nullptr;
876  triton::uint64 address = mem.getAddress();
877  triton::uint32 size = mem.getSize();
878  triton::uint8 concreteValue[triton::size::dqqword] = {0};
879  triton::uint512 value = this->architecture->getConcreteMemoryValue(mem);
880 
881  triton::utils::fromUintToBuffer(value, concreteValue);
882 
883  /*
884  * Symbolic optimization
885  * If the memory access is aligned, don't split the memory.
886  */
887  if (this->modes->isModeEnabled(triton::modes::ALIGNED_MEMORY) && this->isAlignedMemory(address, size)) {
888  return this->getAlignedMemory(address, size)->getAst();
889  }
890 
891  /* If the memory access is 1 byte long, just return the appropriate 8-bit vector */
892  if (size == 1) {
893  const SharedSymbolicExpression& symMem = this->getSymbolicMemory(address);
894  if (symMem) return this->astCtxt->reference(symMem);
895  else return this->astCtxt->bv(concreteValue[size - 1], bitsize::byte);
896  }
897 
898  /* If the memory access is more than 1 byte long, concatenate each memory cell */
899  opVec.reserve(size);
900  while (size) {
901  const SharedSymbolicExpression& symMem = this->getSymbolicMemory(address + size - 1);
902  if (symMem) opVec.push_back(this->astCtxt->reference(symMem));
903  else opVec.push_back(this->astCtxt->bv(concreteValue[size - 1], bitsize::byte));
904  size--;
905  }
906  return this->astCtxt->concat(opVec);
907  }

970  /* Returns the new symbolic memory expression */
971  const SharedSymbolicExpression& SymbolicEngine::createSymbolicMemoryExpression(triton::arch::Instruction& inst, const triton::ast::SharedAbstractNode& node, const triton::arch::MemoryAccess& mem, const std::string& comment) {
972  std::vector<triton::ast::SharedAbstractNode> ret;
973  triton::ast::SharedAbstractNode tmp = nullptr;
974  SharedSymbolicExpression se = nullptr;
975  triton::uint64 address = mem.getAddress();
976  triton::uint32 writeSize = mem.getSize();
977  triton::usize id = this->uniqueSymExprId;
978 
979  std::stringstream s;
980  s << comment << (comment.empty() ? "" : " - ") << inst;
981 
982  /* Record the aligned memory for a symbolic optimization */
983  if (this->modes->isModeEnabled(triton::modes::ALIGNED_MEMORY)) {
984  const SharedSymbolicExpression& aligned = this->newSymbolicExpression(node, MEMORY_EXPRESSION, "Aligned Byte reference - " + s.str());
985  this->addAlignedMemory(address, writeSize, aligned);
986  }
987 
988  /*
989  * As the x86's memory can be accessed without alignment, each byte of the
990  * memory must be assigned to an unique reference.
991  */
992  ret.reserve(mem.getSize());
993  while (writeSize) {
994  triton::uint32 high = ((writeSize * bitsize::byte) - 1);
995  triton::uint32 low = ((writeSize * bitsize::byte) - bitsize::byte);
996  /* Extract each byte of the memory */
997  tmp = this->astCtxt->extract(high, low, node);
998  /* Assign each byte to a new symbolic expression */
999  se = this->newSymbolicExpression(tmp, MEMORY_EXPRESSION, "Byte reference - " + s.str());
1000  /* Set the origin of the symbolic expression */
1001  se->setOriginMemory(triton::arch::MemoryAccess(((address + writeSize) - 1), triton::size::byte));
1002  /* ret is the for the final expression */
1003  ret.push_back(tmp);
1004  /* Assign memory with little endian */
1005  this->addMemoryReference((address + writeSize) - 1, se);
1006  /* continue */
1007  writeSize--;
1008  }
1009 
1010  /* Set implicit read of the base and index registers from an effective address */
1011  this->setImplicitReadRegisterFromEffectiveAddress(inst, mem);
1012 
1013  /* Set explicit write of the memory access */
1014  inst.setStoreAccess(mem, node);
1015 
1016  /* If there is only one reference, we return the symbolic expression */
1017  if (ret.size() == 1) {
1018  /* Synchronize the concrete state */
1019  this->architecture->setConcreteMemoryValue(mem, tmp->evaluate());
1020  return this->addSymbolicExpressions(inst, id);
1021  }
1022 
1023  /* Otherwise, we return the concatenation of all symbolic expressions */
1024  tmp = this->astCtxt->concat(ret);
1025 
1026  /* Synchronize the concrete state */
1027  this->architecture->setConcreteMemoryValue(mem, tmp->evaluate());
1028 
1029  se = this->newSymbolicExpression(tmp, MEMORY_EXPRESSION, "Temporary concatenation reference - " + s.str());
1030  se->setOriginMemory(triton::arch::MemoryAccess(address, mem.getSize()));
1031 
1032  return this->addSymbolicExpressions(inst, id);
1033  }

1036  /* Returns the parent AST after inserting the subregister (node) in its AST. */
1037  triton::ast::SharedAbstractNode SymbolicEngine::insertSubRegisterInParent(const triton::arch::Register& reg, const triton::ast::SharedAbstractNode& node, bool zxForAssign) {
1038  const triton::arch::Register& parentReg = this->architecture->getParentRegister(reg);
1039 
1040  /* If it's a flag register, there is nothing to do with sub register */
1041  if (this->architecture->isFlag(reg)) {
1042  return node;
1043  }
1044 
1045  switch (reg.getSize()) {
1046  /* ----------------------------------------------------------------*/
1047  case triton::size::byte: {
1048  const auto& origReg = this->getRegisterAst(parentReg);
1049  /*
1050  * Mainly used for x86
1051  * r[........xxxxxxxx]
1052  */
1053  if (reg.getLow() == 0) {
1054  const auto& keep = this->astCtxt->extract((parentReg.getBitSize() - 1), bitsize::byte, origReg);
1055  return this->astCtxt->concat(keep, node);
1056  }
1057  /*
1058  * Mainly used for x86
1059  * r[xxxxxxxx........]
1060  */
1061  else {
1062  const auto& keep = this->astCtxt->extract((parentReg.getBitSize() - 1), bitsize::word, origReg);
1063  return this->astCtxt->concat(keep, this->astCtxt->concat(node, this->astCtxt->extract((bitsize::byte - 1), 0, origReg)));
1064  }
1065  }
1066 
1067  /* ----------------------------------------------------------------*/
1068  case triton::size::word: {
1069  const auto& origReg = this->getRegisterAst(parentReg);
1070  return this->astCtxt->concat(this->astCtxt->extract((parentReg.getBitSize() - 1), bitsize::word, origReg), node);
1071  }
1072 
1073  /* ----------------------------------------------------------------*/
1074  case triton::size::dword:
1075  case triton::size::qword:
1076  case triton::size::dqword:
1077  case triton::size::qqword:
1078  case triton::size::dqqword: {
1079  if (zxForAssign == false) {
1080  if (parentReg.getBitSize() > reg.getBitSize()) {
1081  const auto& origReg = this->getRegisterAst(parentReg);
1082  return this->astCtxt->concat(this->astCtxt->extract((parentReg.getBitSize() - 1), reg.getHigh() + 1, origReg), node);
1083  }
1084  else {
1085  return node;
1086  }
1087  }
1088  /* zxForAssign == true */
1089  else {
1090  return this->astCtxt->zx(parentReg.getBitSize() - node->getBitvectorSize(), node);
1091  }
1092  }
1093  /* ----------------------------------------------------------------*/
1094  }
1095 
1096  throw triton::exceptions::SymbolicEngine("SymbolicEngine::insertSubRegisterInParent(): Invalid register size.");
1097  }

1134  /* Assigns a symbolic expression to a register */
1135  void SymbolicEngine::assignSymbolicExpressionToRegister(const SharedSymbolicExpression& se, const triton::arch::Register& reg) {

1138

1142  }

1162  void SymbolicEngine::assignSymbolicExpressionToMemory(const SharedSymbolicExpression& se, const triton::arch::MemoryAccess& mem) {

1170  }

1174  this->addAlignedMemory(address, writeSize, se);
1175  }
1176

1178  * As the x86's memory can be accessed without alignment, each byte of the
1179  * memory must be assigned to an unique reference.
1180  */

1183  triton::uint32 low = ((writeSize * bitsize::byte) - bitsize::byte);

1193  writeSize--;
1194  }

1256  /* Initializes the memory access AST (LOAD and STORE) */
1257  void SymbolicEngine::initLeaAst(triton::arch::MemoryAccess& mem, bool force) {
1258  if (mem.getBitSize() >= bitsize::byte) {
1259  const triton::arch::Register& base = mem.getConstBaseRegister();
1260  const triton::arch::Register& index = mem.getConstIndexRegister();
1261  const triton::arch::Register& seg = mem.getConstSegmentRegister();
1262  triton::uint64 scaleValue = mem.getConstScale().getValue();
1263  triton::uint64 dispValue = mem.getConstDisplacement().getValue();
1264  triton::uint32 bitSize = (this->architecture->isRegisterValid(base) ? base.getBitSize() :
1265  (this->architecture->isRegisterValid(index) ? index.getBitSize() :
1266  (mem.getConstDisplacement().getBitSize() ? mem.getConstDisplacement().getBitSize() :
1267  this->architecture->gprBitSize()
1268  )
1269  )
1270  );
1271 
1272  /* Initialize the AST of the memory access (LEA) -> ((pc + base) + (index * scale) + disp) */
1273  auto pcPlusBaseAst = mem.getPcRelative() ? this->astCtxt->bv(mem.getPcRelative(), bitSize) :
1274  (this->architecture->isRegisterValid(base) ? this->getRegisterAst(base) :
1275  this->astCtxt->bv(0, bitSize));
1276 
1277  auto indexMulScaleAst = this->astCtxt->bvmul(
1278  (this->architecture->isRegisterValid(index) ? this->getRegisterAst(index) : this->astCtxt->bv(0, bitSize)),
1279  this->astCtxt->bv(scaleValue, bitSize)
1280  );
1281 
1282  auto dispAst = this->astCtxt->bv(dispValue, bitSize);
1283  auto leaAst = this->astCtxt->bvadd(
1284  index.isSubtracted() ? this->astCtxt->bvsub(pcPlusBaseAst, indexMulScaleAst) : this->astCtxt->bvadd(pcPlusBaseAst, indexMulScaleAst),
1285  dispAst
1286  );
1287 
1288  /* Use segments as base address instead of selector into the GDT. */
1289  if (this->architecture->isRegisterValid(seg)) {
1290  leaAst = this->astCtxt->bvadd(
1291  this->getRegisterAst(seg),
1292  this->astCtxt->sx((seg.getBitSize() - bitSize), leaAst)
1293  );
1294  }
1295 
1296  /* Set AST */
1297  mem.setLeaAst(leaAst);
1298 
1299  /* Initialize the address only if it is not already defined */
1300  if (!mem.getAddress() || force)
1301  mem.setAddress(leaAst->evaluate().convert_to<triton::uint64>());
1302  }
