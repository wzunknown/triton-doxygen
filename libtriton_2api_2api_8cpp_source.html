534  void API::initEngines(void) {
535  this->checkArchitecture();
536 
537  this->symbolic = new(std::nothrow) triton::engines::symbolic::SymbolicEngine(&this->arch, this->modes, this->astCtxt, &this->callbacks);
538  if (this->symbolic == nullptr)
539  throw triton::exceptions::API("API::initEngines(): Not enough memory.");
540 
541  this->solver = new(std::nothrow) triton::engines::solver::SolverEngine();
542  if (this->solver == nullptr)
543  throw triton::exceptions::API("API::initEngines(): Not enough memory.");
544 
545  this->taint = new(std::nothrow) triton::engines::taint::TaintEngine(this->modes, this->symbolic, *this->getCpuInstance());
546  if (this->taint == nullptr)
547  throw triton::exceptions::API("API::initEngines(): Not enough memory.");
548 
549  this->irBuilder = new(std::nothrow) triton::arch::IrBuilder(&this->arch, this->modes, this->astCtxt, this->symbolic, this->taint);
550  if (this->irBuilder == nullptr)
551  throw triton::exceptions::API("API::initEngines(): Not enough memory.");
552 
553  /* Setup registers shortcut */
554  this->registers.init(this->arch.getArchitecture());
555  }
