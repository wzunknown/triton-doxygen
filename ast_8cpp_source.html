424  void BvashrNode::init(bool withParents) {
425  triton::uint32 shift = 0;
426  triton::uint512 mask = 0;
427  triton::uint512 value = 0;
428 
429  if (this->children.size() < 2)
430  throw triton::exceptions::Ast("BvashrNode::init(): Must take at least two children.");
431 
432  if (this->children[0]->getBitvectorSize() != this->children[1]->getBitvectorSize())
433  throw triton::exceptions::Ast("BvashrNode::init(): Must take two nodes of same size.");
434 
435  value = this->children[0]->evaluate();
436  shift = this->children[1]->evaluate().convert_to<triton::uint32>();
437 
438  /* Init attributes */
439  this->size = this->children[0]->getBitvectorSize();
440  this->level = 1;
441  this->symbolized = false;
442 
443  /* Mask based on the sign */
444  if (this->children[0]->isSigned()) {
445  mask = 1;
446  mask = ((mask << (this->size-1)) & this->getBitvectorMask());
447  }
448 
449  if (shift >= this->size && this->children[0]->isSigned()) {
450  this->eval = -1;
451  this->eval &= this->getBitvectorMask();
452  }
453 
454  else if (shift >= this->size && !this->children[0]->isSigned()) {
455  this->eval = 0;
456  }
457 
458  else if (shift == 0) {
459  this->eval = value;
460  }
461 
462  else {
463  this->eval = value & this->getBitvectorMask();
464  for (triton::uint32 index = 0; index < shift; index++) {
465  this->eval = (((this->eval >> 1) | mask) & this->getBitvectorMask());
466  }
467  }
468 
469  /* Init children and spread information */
470  for (triton::uint32 index = 0; index < this->children.size(); index++) {
471  this->children[index]->setParent(this);
472  this->symbolized |= this->children[index]->isSymbolized();
473  this->level = std::max(this->children[index]->getLevel() + 1, this->level);
474  }
475 
476  /* Init parents if needed */
477  if (withParents) {
478  this->initParents();
479  }
480 
481  this->initHash();
482  }
