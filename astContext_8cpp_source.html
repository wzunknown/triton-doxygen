46  SharedAbstractNode AstContext::collect(const SharedAbstractNode& node) {
47  /*
48  * We keep references to nodes which belong to a depth in the AST which is
49  * a multiple of 10000. Thus, when the root node is destroyed, the stack recursivity
50  * stops when the depth level of 10000 is reached, because the nodes there still
51  * have a reference to them in the AST manager. The destruction will continue at the
52  * next allocation of nodes and so on. So, it means that ASTs are destroyed by steps
53  * of depth of 10000 which avoids the overflow while keeping a good scale.
54  *
55  * See: #753.
56  */
57  triton::uint32 lvl = node->getLevel();
58  if (lvl != 0 && (lvl % 10000) == 0) {
59  this->nodes.push_front(node);
60  }
61  return node;
62  }

373  SharedAbstractNode AstContext::bvrol(const SharedAbstractNode& expr, const SharedAbstractNode& rot) {
374  /*
375  * If the mode SYMBOLIZE_INDEX_ROTATION we apply a AST transformation
376  * in order to make index rotation symbolic. Note that this mode increases the
377  * complexity of solving.
378  *
379  * bvrol(rot, expr) = ((expr << (rot % size)) | (expr >> (size - (rot % size))))
380  **/
381  if (this->modes->isModeEnabled(triton::modes::SYMBOLIZE_INDEX_ROTATION)) {
382  auto size = expr->getBitvectorSize();
383  auto bvsize = this->bv(size, size);
384  const auto& node = this->bvor(
385  this->bvshl(expr, this->bvsmod(rot, bvsize)),
386  this->bvlshr(expr, this->bvsub(bvsize, this->bvsmod(rot, bvsize)))
387  );
388  return node;
389  }
390

392  SharedAbstractNode node = std::make_shared<BvrolNode>(expr, this->integer(rot->evaluate()));

1072  SharedAbstractNode AstContext::simplify_concat(std::vector<SharedAbstractNode> exprs) {
1073  /*
1074  * Optimization: concatenate extractions in one if possible. We are
1075  * trying to find out whether this is a concatenation of sequential
1076  * extractions from the same AST. Thus, we can make only one extraction.
1077  *
1078  * (concat ((_ extract 31 24) a) ((_ extract 23 16) a)
1079  * ((_ extract 15 8) a) ((_ extract 7 0) a)) =>
1080  * ((_ extract 31 0) a)
1081  *
1082  * So, we are examining whether we can replace concatenation with one
1083  * extraction ((_ extract high low) ast_ref).
1084  **/
1085  uint32 high = 0; // target extraction upper bound
1086  uint32 low = 1; // target extraction lower bound
1087  SharedAbstractNode ast_ref = 0; // target AST to extract from
1088  SharedAbstractNode ast_node = 0; // ast_ref with unrolled references
1089 
1090  /* Try to join all extractions into one from the right to the left */
1091  while (!exprs.empty()) {
1092  /* Get the right most node */
1093  auto n = exprs.back();
1094  exprs.pop_back();
1095 
1096  /* Returns the first non referene node encountered */
1097  n = triton::ast::dereference(n);
1098 
1099  if (n->getType() == CONCAT_NODE) {
1100  /* Append concatenation children to the right */
1101  for (const auto& part : n->getChildren()) {
1102  exprs.push_back(part);
1103  }
1104  continue;
1105  }
1106 
1107  if (n->getType() != EXTRACT_NODE) {
1108  /* We cannot optimize if at least one node is not an extract */
1109  return 0;
1110  }
1111 
1112  /* Get extraction arguments */
1113  const auto& childs = n->getChildren();
1114  auto hi = reinterpret_cast<IntegerNode*>(childs[0].get())->getInteger().convert_to<uint32>();
1115  auto lo = reinterpret_cast<IntegerNode*>(childs[1].get())->getInteger().convert_to<uint32>();
1116  if (hi < lo) {
1117  return 0;
1118  }
1119  n = childs[2];
1120 
1121  /* Returns the first non referene node encountered */
1122  n = triton::ast::dereference(n);
1123 
1124  if (!ast_ref) {
1125  /* First found extraction node */
1126  high = hi;
1127  low = lo;
1128  ast_ref = childs[2];
1129  ast_node = n;
1130  continue;
1131  }
1132 
1133  /*
1134  * Check that unrolled target AST is equal to extraction node child.
1135  * Also, check that extraction lower bound connects with target
1136  * extraction upper bound.
1137  **/
1138  if (high + 1 != lo || !n->equalTo(ast_node)) {
1139  return 0;
1140  }
1141  high = hi;
1142  }
1143 
1144  if (high < low) {
1145  return 0;
1146  }
1147 
1148  /* Perform target extraction */
1149  return this->extract(high, low, ast_ref);
1150  }

1153  SharedAbstractNode AstContext::simplify_extract(triton::uint32 high, triton::uint32 low, const SharedAbstractNode& expr) {
1154  auto size = expr->getBitvectorSize();
1155 
1156  if (high <= low || high >= size) {
1157  return 0;
1158  }
1159 
1160  auto node = expr;
1161  while (true) {
1162  /* Returns the first non referene node encountered */
1163  auto n = triton::ast::dereference(node);
1164 
1165  if (n->getType() == CONCAT_NODE) {
1166  /*
1167  * Optimization: If we extract the full part of concatenation, just
1168  * return the part. We are trying to find a part of concatenation
1169  * that we can extract from. Thus, we can extract from only one part
1170  * of concatenation.
1171  *
1172  * ((_ extract 11 9) (concat (_ bv1 8) (_ bv2 8) (_ bv3 8) (_ bv4 8))) =>
1173  * ((_ extract 3 1) (_ bv3 8))
1174  */
1175  auto hi = n->getBitvectorSize() - 1;
1176  bool found = false;
1177  /* Search for part of concatenation we can extract from. Iterate
1178  * from the left to the right. */
1179  for (const auto& part : n->getChildren()) {
1180  if (hi < high) {
1181  /* Did not find a part we can extract from */
1182  break;
1183  }
1184  auto sz = part->getBitvectorSize();
1185  auto lo = hi + 1 - sz;
1186  if (hi == high && lo == low) {
1187  /* We are extracting the full part, just return it */
1188  return part;
1189  }
1190  if (hi >= high && lo <= low) {
1191  /* Extract from part: ((_ extract high-lo low-lo) part) */
1192  node = part;
1193  high -= lo;
1194  low -= lo;
1195  found = true;
1196  break;
1197  }
1198  hi -= sz;
1199  }
1200  if (found) {
1201  /* Optimize ((_ extract high low) node) one more time */
1202  continue;
1203  }
1204  }
1205  else if (n->getType() == ZX_NODE || n->getType() == SX_NODE) {
1206  /*
1207  * Optimization: If we extract from the node being extended, just
1208  * return the node
1209  *
1210  * ((_ extract 31 0) ((_ zero_extend 32) (_ bv1 32))) => (_ bv1 32)
1211  *
1212  * ((_ extract 7 0) ((_ sign_extend 32) (_ bv1 32))) =>
1213  * ((_ extract 7 0) (_ bv1 32))
1214  **/
1215  n = n->getChildren()[1];
1216  auto sz = n->getBitvectorSize();
1217  if (low == 0 && high + 1 == sz) {
1218  /* Just return the node being extended */
1219  return n;
1220  }
1221  if (high < sz) {
1222  /* Optimize ((_ extract high low) n) one more time */
1223  node = n;
1224  continue;
1225  }
1226  }
1227  break;
1228  }
1229 
1230  /* Returns the first non referene node encountered */
1231  auto n = triton::ast::dereference(node);
1232 
1233  /*
1234  * Optimization: extract from extract is one extract
1235  *
1236  * ((_ extract high low) ((_ extract hi lo) a)) =>
1237  * ((_ extract high+lo low+lo) a)
1238  **/
1239  if (n->getType() == EXTRACT_NODE) {
1240  const auto& childs = n->getChildren();
1241  auto hi = reinterpret_cast<IntegerNode*>(childs[0].get())->getInteger().convert_to<uint32>();
1242  auto lo = reinterpret_cast<IntegerNode*>(childs[1].get())->getInteger().convert_to<uint32>();
1243  if (lo + high <= hi) {
1244  node = childs[2];
1245  high += lo;
1246  low += lo;
1247  }
1248  }
1249 
1250  return node == expr ? 0 : this->extract(high, low, node);
1251  }
