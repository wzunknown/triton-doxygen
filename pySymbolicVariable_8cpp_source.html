227  static PyObject* SymbolicVariable_richcompare(PyObject* self, PyObject* other, int op) {
228  PyObject* result = nullptr;
229  triton::usize id1 = 0;
230  triton::usize id2 = 0;
231 
232  if (!PySymbolicVariable_Check(other)) {
233  result = Py_NotImplemented;
234  }
235  else {
236  id1 = PySymbolicVariable_AsSymbolicVariable(self)->getId();
237  id2 = PySymbolicVariable_AsSymbolicVariable(other)->getId();
238 
239  switch (op) {
240  case Py_LT:
241  result = (id1 < id2) ? Py_True : Py_False;
242  break;
243  case Py_LE:
244  result = (id1 <= id2) ? Py_True : Py_False;
245  break;
246  case Py_EQ:
247  result = (id1 == id2) ? Py_True : Py_False;
248  break;
249  case Py_NE:
250  result = (id1 != id2) ? Py_True : Py_False;
251  break;
252  case Py_GT:
253  result = (id1 > id2) ? Py_True : Py_False;
254  break;
255  case Py_GE:
256  result = (id1 >= id2) ? Py_True : Py_False;
257  break;
258  }
259  }
260 
261  Py_INCREF(result);
262  return result;
263  }

346  PyObject* PySymbolicVariable(const triton::engines::symbolic::SharedSymbolicVariable& symVar) {
347  if (symVar == nullptr) {
348  Py_INCREF(Py_None);
349  return Py_None;
350  }
351 
352  PyType_Ready(&SymbolicVariable_Type);
353  // Build the new object the python way (calling operator() on the type) as
354  // it crash otherwise (certainly due to incorrect shared_ptr initialization).
355  auto* object = (triton::bindings::python::SymbolicVariable_Object*)PyObject_CallObject((PyObject*)&SymbolicVariable_Type, nullptr);
356  if (object != NULL) {
357  object->symVar = symVar;
358  }
359 
360  return (PyObject*)object;
361  }
