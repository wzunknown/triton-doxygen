357  static PyObject* Instruction_getOperands(PyObject* self, PyObject* noarg) {
358  try {
359  triton::arch::Instruction* inst;
360  triton::usize opSize;
361  PyObject* operands;
362 
363  inst = PyInstruction_AsInstruction(self);
364  opSize = inst->operands.size();
365  operands = xPyList_New(opSize);
366 
367  for (triton::usize index = 0; index < opSize; index++) {
368  PyObject* obj = nullptr;
369 
370  if (inst->operands[index].getType() == triton::arch::OP_IMM) {
371  const triton::arch::Immediate& imm = inst->operands[index].getConstImmediate();
372  obj = PyImmediate(imm);
373  }
374  else if (inst->operands[index].getType() == triton::arch::OP_MEM) {
375  const triton::arch::MemoryAccess& mem = inst->operands[index].getConstMemory();
376  obj = PyMemoryAccess(mem);
377  }
378  else if (inst->operands[index].getType() == triton::arch::OP_REG) {
379  const triton::arch::Register& reg = inst->operands[index].getConstRegister();
380  obj = PyRegister(reg);
381  }
382  else
383  continue;
384 
385  PyList_SetItem(operands, index, obj);
386  }
387 
388  return operands;
389  }
390  catch (const triton::exceptions::Exception& e) {
391  return PyErr_Format(PyExc_TypeError, "%s", e.what());
392  }
