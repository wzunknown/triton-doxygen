42  std::vector<std::unordered_map<triton::usize, SolverModel>> Z3Solver::getModels(const triton::ast::SharedAbstractNode& node, triton::uint32 limit, triton::engines::solver::status_e* status) const {
43  std::vector<std::unordered_map<triton::usize, SolverModel>> ret;
44  triton::ast::SharedAbstractNode onode = node;
45  triton::ast::TritonToZ3Ast z3Ast{false};
46 
47  try {
48  if (onode == nullptr)
49  throw triton::exceptions::SolverEngine("Z3Solver::getModels(): node cannot be null.");
50 
51  /* Z3 does not need an assert() as root node */
52  if (node->getType() == triton::ast::ASSERT_NODE)
53  onode = node->getChildren()[0];
54 
55  if (onode->isLogical() == false)
56  throw triton::exceptions::SolverEngine("Z3Solver::getModels(): Must be a logical node.");
57 
58  z3::expr expr = z3Ast.convert(onode);
59  z3::context& ctx = expr.ctx();
60  z3::solver solver(ctx);
61 
62  /* Create a solver and add the expression */
63  solver.add(expr);
64 
65  z3::params p(ctx);
66 
67  /* Define the timeout */
68  if (this->timeout) {
69  p.set(":timeout", this->timeout);
70  }
71 
72  /* Define memory limit */
73  if (this->memoryLimit) {
74  p.set(":max_memory", this->memoryLimit);
75  }
76 
77  solver.set(p);
78 
79  /* Get first model */
80  z3::check_result res = solver.check();
81 
82  /* Write back the status code of the first constraint */
83  this->writeBackStatus(solver, res, status);
84 
85  /* Check if it is sat */
86  for (; res == z3::sat && limit >= 1; res = solver.check()) {
87 
88  /* Get model */
89  z3::model m = solver.get_model();
90 
91  /* Traversing the model */
92  std::unordered_map<triton::usize, SolverModel> smodel;
93  z3::expr_vector args(ctx);
94  for (triton::uint32 i = 0; i < m.size(); i++) {
95 
96  /* Get the z3 variable */
97  z3::func_decl z3Variable = m[i];
98 
99  /* Get the name as std::string from a z3 variable */
100  std::string varName = z3Variable.name().str();
101 
102  /* Get z3 expr */
103  z3::expr exp = m.get_const_interp(z3Variable);
104 
105  /* Get the size of a z3 expr */
106  triton::uint32 bvSize = exp.get_sort().bv_size();
107 
108  /* Get the value of a z3 expr */
109  std::string svalue = Z3_get_numeral_string(ctx, exp);
110 
111  /* Convert a string value to a integer value */
112  triton::uint512 value = triton::uint512(svalue);
113 
114  /* Create a triton model */
115  SolverModel trionModel = SolverModel(z3Ast.variables[varName], value);
116 
117  /* Map the result */
118  smodel[trionModel.getId()] = trionModel;
119 
120  /* Uniq result */
121  if (exp.get_sort().is_bv())
122  args.push_back(ctx.bv_const(varName.c_str(), bvSize) != ctx.bv_val(svalue.c_str(), bvSize));
123 
124  }
125 
126  /* Escape last models */
127  solver.add(triton::engines::solver::mk_or(args));
128 
129  /* If there is model available */
130  if (smodel.size() > 0)
131  ret.push_back(smodel);
132 
133  /* Decrement the limit */
134  limit--;
135  }
136  }
137  catch (const z3::exception& e) {
138  throw triton::exceptions::SolverEngine(std::string("Z3Solver::getModels(): ") + e.msg());
139  }
140 
141  return ret;
142  }

247  void Z3Solver::writeBackStatus(z3::solver& solver, z3::check_result res, triton::engines::solver::status_e* status) const {
248  if (status != nullptr) {
249  switch (res) {
250  case z3::sat:
251  *status = triton::engines::solver::SAT;
252  break;
253 
254  case z3::unsat:
255  *status = triton::engines::solver::UNSAT;
256  break;
257 
258  case z3::unknown:
259  if (solver.reason_unknown() == "timeout") {
260  *status = triton::engines::solver::TIMEOUT;
261  }
262  else if (solver.reason_unknown() == "max. memory exceeded")
263  {
264  *status = triton::engines::solver::OUTOFMEM;
265  }
266  else {
267  *status = triton::engines::solver::UNKNOWN;
268  }
269  break;
270  }
