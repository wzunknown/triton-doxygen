358  void x86Cpu::disassembly(triton::arch::Instruction& inst) const {
359  triton::extlibs::capstone::cs_insn* insn;
360  triton::usize count = 0;
361 
362  /* Check if the opcode and opcode' size are defined */
363  if (inst.getOpcode() == nullptr || inst.getSize() == 0)
364  throw triton::exceptions::Disassembly("x86Cpu::disassembly(): Opcode and opcodeSize must be definied.");
365 
366  /* Clear instructicon's operands if alredy defined */
367  inst.operands.clear();
368 
369  /* Update instruction address if undefined */
370  if (!inst.getAddress()) {
371  inst.setAddress(this->getConcreteRegisterValue(this->getProgramCounter()).convert_to<triton::uint64>());
372  }
373 
374  /* Let's disass and build our operands */
375  count = triton::extlibs::capstone::cs_disasm(this->handle, inst.getOpcode(), inst.getSize(), inst.getAddress(), 0, &insn);
376  if (count > 0) {
377  /* Detail information */
378  triton::extlibs::capstone::cs_detail* detail = insn->detail;
379 
380  /* Init the disassembly */
381  std::stringstream str;
382 
383  str << insn[0].mnemonic;
384  if (detail->x86.op_count)
385  str << " " << insn[0].op_str;
386 
387  inst.setDisassembly(str.str());
388 
389  /* Refine the size */
390  inst.setSize(insn[0].size);
391 
392  /* Init the instruction's type */
393  inst.setType(this->capstoneInstructionToTritonInstruction(insn[0].id));
394 
395  /* Init the instruction's prefix */
396  inst.setPrefix(this->capstonePrefixToTritonPrefix(detail->x86.prefix[0]));
397 
398  /* Init operands */
399  for (triton::uint32 n = 0; n < detail->x86.op_count; n++) {
400  triton::extlibs::capstone::cs_x86_op* op = &(detail->x86.operands[n]);
401  switch(op->type) {
402 
403  case triton::extlibs::capstone::X86_OP_IMM:
404  inst.operands.push_back(triton::arch::OperandWrapper(triton::arch::Immediate(op->imm, op->size)));
405  break;
406 
407  case triton::extlibs::capstone::X86_OP_MEM: {
408  triton::arch::MemoryAccess mem;
409 
410  /* Set the size of the memory access */
411  mem.setBits(((op->size * triton::bitsize::byte) - 1), 0);
412 
413  /* LEA if exists */
414  const triton::arch::Register segment(*this, this->capstoneRegisterToTritonRegister(op->mem.segment));
415  const triton::arch::Register base(*this, this->capstoneRegisterToTritonRegister(op->mem.base));
416  const triton::arch::Register index(*this, this->capstoneRegisterToTritonRegister(op->mem.index));
417 
418  triton::uint32 immsize = (
419  this->isRegisterValid(base.getId()) ? base.getSize() :
420  this->isRegisterValid(index.getId()) ? index.getSize() :
421  this->gprSize()
422  );
423 
424  triton::arch::Immediate disp(op->mem.disp, immsize);
425  triton::arch::Immediate scale(op->mem.scale, immsize);
426 
427  /* Specify that LEA contains a PC relative */
428  if (base.getId() == this->pcId)
429  mem.setPcRelative(inst.getNextAddress());
430 
431  mem.setSegmentRegister(segment);
432  mem.setBaseRegister(base);
433  mem.setIndexRegister(index);
434  mem.setDisplacement(disp);
435  mem.setScale(scale);
436 
437  inst.operands.push_back(triton::arch::OperandWrapper(mem));
438  break;
439  }
440 
441  case triton::extlibs::capstone::X86_OP_REG:
442  inst.operands.push_back(triton::arch::OperandWrapper(triton::arch::Register(*this, this->capstoneRegisterToTritonRegister(op->reg))));
443  break;
444 
445  default:
446  break;
447  }
448 
449  }
450  /* Set branch */
451  if (detail->groups_count > 0) {
452  for (triton::uint32 n = 0; n < detail->groups_count; n++) {
453  if (detail->groups[n] == triton::extlibs::capstone::X86_GRP_JUMP)
454  inst.setBranch(true);
455  if (detail->groups[n] == triton::extlibs::capstone::X86_GRP_JUMP ||
456  detail->groups[n] == triton::extlibs::capstone::X86_GRP_CALL ||
457  detail->groups[n] == triton::extlibs::capstone::X86_GRP_RET)
458  inst.setControlFlow(true);
459  }
460  }
461  triton::extlibs::capstone::cs_free(insn, count);
462  }
463  else
464  throw triton::exceptions::Disassembly("x86Cpu::disassembly(): Failed to disassemble the given code.");
465  }
