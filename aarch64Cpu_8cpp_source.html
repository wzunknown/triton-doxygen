326  void AArch64Cpu::disassembly(triton::arch::Instruction& inst) const {
327  triton::extlibs::capstone::cs_insn* insn;
328  triton::usize count = 0;
329  triton::uint32 size = 0;
330 
331  /* Check if the opcode and opcode' size are defined */
332  if (inst.getOpcode() == nullptr || inst.getSize() == 0)
333  throw triton::exceptions::Disassembly("AArch64Cpu::disassembly(): Opcode and opcodeSize must be definied.");
334 
335  /* Clear instructicon's operands if alredy defined */
336  inst.operands.clear();
337 
338  /* Update instruction address if undefined */
339  if (!inst.getAddress()) {
340  inst.setAddress(this->getConcreteRegisterValue(this->getProgramCounter()).convert_to<triton::uint64>());
341  }
342 
343  /* Let's disass and build our operands */
344  count = triton::extlibs::capstone::cs_disasm(this->handle, inst.getOpcode(), inst.getSize(), inst.getAddress(), 0, &insn);
345  if (count > 0) {
346  /* Detail information */
347  triton::extlibs::capstone::cs_detail* detail = insn->detail;
348 
349  /* Init the disassembly */
350  std::stringstream str;
351 
352  str << insn[0].mnemonic;
353  if (detail->arm64.op_count)
354  str << " " << insn[0].op_str;
355 
356  inst.setDisassembly(str.str());
357 
358  /* Refine the size */
359  inst.setSize(insn[0].size);
360 
361  /* Init the instruction's type */
362  inst.setType(this->capstoneInstructionToTritonInstruction(insn[0].id));
363 
364  /* Init the instruction's code codition */
365  inst.setCodeCondition(this->capstoneConditionToTritonCondition(detail->arm64.cc));
366 
367  /* Init the instruction's write back flag */
368  inst.setWriteBack(detail->arm64.writeback);
369 
370  /* Set True if the instruction udpate flags */
371  inst.setUpdateFlag(detail->arm64.update_flags);
372 
373  /* Init operands */
374  for (triton::uint32 n = 0; n < detail->arm64.op_count; n++) {
375  triton::extlibs::capstone::cs_arm64_op* op = &(detail->arm64.operands[n]);
376  switch(op->type) {
377 
378  case triton::extlibs::capstone::ARM64_OP_IMM: {
379  triton::arch::Immediate imm(op->imm, size ? size : triton::size::qword);
380 
381  /*
382  * Instruction such that CBZ, CBNZ or TBZ may imply a wrong size.
383  * So, if Triton truncates the value by setting a size less than
384  * the original one, we redefine the size automatically.
385  */
386  if (static_cast<triton::uint64>(op->imm) > imm.getValue()) {
387  imm = Immediate();
388  imm.setValue(op->imm, 0); /* By setting 0 as size, we automatically identify the size of the value */
389  }
390 
391  /* Set Shift type and value */
392  imm.setShiftType(this->capstoneShiftToTritonShift(op->shift.type));
393  imm.setShiftValue(op->shift.value);
394 
395  inst.operands.push_back(triton::arch::OperandWrapper(imm));
396  break;
397  }
398 
399  case triton::extlibs::capstone::ARM64_OP_MEM: {
400  triton::arch::MemoryAccess mem;
401 
402  /* Set the size of the memory access */
403  mem.setBits(size ? ((size * triton::bitsize::byte) - 1) : triton::bitsize::qword - 1, 0);
404 
405  /* LEA if exists */
406  triton::arch::Register base(*this, this->capstoneRegisterToTritonRegister(op->mem.base));
407  triton::arch::Register index(*this, this->capstoneRegisterToTritonRegister(op->mem.index));
408 
409  triton::uint32 immsize = (
410  this->isRegisterValid(base.getId()) ? base.getSize() :
411  this->isRegisterValid(index.getId()) ? index.getSize() :
412  this->gprSize()
413  );
414 
415  triton::arch::Immediate disp(op->mem.disp, immsize);
416 
417  /* Specify that LEA contains a PC relative */
418  /* FIXME: Valid in ARM64 ? */
419  if (base.getId() == this->pcId) {
420  mem.setPcRelative(inst.getNextAddress());
421  }
422 
423  /* Set Shift type and value */
424  index.setShiftType(this->capstoneShiftToTritonShift(op->shift.type));
425  index.setShiftValue(op->shift.value);
426 
427  /* Set extend type and size */
428  index.setExtendType(this->capstoneExtendToTritonExtend(op->ext));
429  if (op->ext != triton::extlibs::capstone::ARM64_EXT_INVALID) {
430  index.setExtendedSize(base.getBitSize());
431  }
432 
433  /* Note that in ARM64 there is no segment register and scale value */
434  mem.setBaseRegister(base);
435  mem.setIndexRegister(index);
436  mem.setDisplacement(disp);
437 
438  /* If there is an index register available, set scale to 1 to perform this following computation (base) + (index * scale) */
439  if (this->isRegisterValid(index.getId())) {
440  mem.setScale(triton::arch::Immediate(1, immsize));
441  }
442 
443  inst.operands.push_back(triton::arch::OperandWrapper(mem));
444  break;
445  }
446 
447  case triton::extlibs::capstone::ARM64_OP_REG: {
448  triton::arch::Register reg(*this, this->capstoneRegisterToTritonRegister(op->reg));
449 
450  /* Set Shift type and value */
451  reg.setShiftType(this->capstoneShiftToTritonShift(op->shift.type));
452  reg.setShiftValue(op->shift.value);
453 
454  /* Set extend type and size */
455  reg.setExtendType(this->capstoneExtendToTritonExtend(op->ext));
456  if (op->ext != triton::extlibs::capstone::ARM64_EXT_INVALID) {
457  reg.setExtendedSize(size * triton::bitsize::byte);
458  }
459 
460  /* Define a base address for next operand */
461  if (!size) {
462  size = reg.getSize();
463  }
464 
465  inst.operands.push_back(triton::arch::OperandWrapper(reg));
466  break;
467  }
468 
469  default:
470  /* NOTE: FP, CIMM, and missing one are not supported yet. */
471  throw triton::exceptions::Disassembly("AArch64Cpu::disassembly(): Invalid operand.");
472  } // switch
473  } // for operand
474 
475  /* Set control flow */
476  if (insn[0].id == triton::extlibs::capstone::ARM64_INS_RET)
477  inst.setControlFlow(true);
478 
479  /* Set branch */
480  if (detail->groups_count > 0) {
481  for (triton::uint32 n = 0; n < detail->groups_count; n++) {
482  if (detail->groups[n] == triton::extlibs::capstone::ARM64_GRP_JUMP) {
483  inst.setBranch(true);
484  inst.setControlFlow(true);
485  }
486  }
487  }
488 
489  /* Free capstone stuffs */
490  triton::extlibs::capstone::cs_free(insn, count);
491  }
492  else
493  throw triton::exceptions::Disassembly("AArch64Cpu::disassembly(): Failed to disassemble the given code.");
494  }

511  triton::uint64 addr = 0;
512  triton::uint32 size = 0;
