312  static PyObject* Register_richcompare(PyObject* self, PyObject* other, int op) {
313  PyObject* result = nullptr;
314  triton::uint32 id1 = 0;
315  triton::uint32 id2 = 0;
316 
317  if (!PyRegister_Check(other)) {
318  result = Py_NotImplemented;
319  }
320 
321  else {
322  id1 = PyRegister_AsRegister(self)->getId();
323  id2 = PyRegister_AsRegister(other)->getId();
324 
325  switch (op) {
326  case Py_LT:
327  result = (id1 < id2) ? Py_True : Py_False;
328  break;
329  case Py_LE:
330  result = (id1 <= id2) ? Py_True : Py_False;
331  break;
332  case Py_EQ:
333  result = (id1 == id2) ? Py_True : Py_False;
334  break;
335  case Py_NE:

337  break;
338  case Py_GT:

340  break;
